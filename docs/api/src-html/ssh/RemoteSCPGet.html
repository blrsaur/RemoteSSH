011/**

013 * SSHConfig Interface Gets files from remote hosts
014 */
015public class RemoteSCPGet  {
016
017        String host = "";
018        String user = "";
019        Integer port=0;
020        String userpass="";
021        String file = "";
022        String localdir = "";
023        String usercommand = "";
024        String output = "";
025        String characterSet="";

069        public RemoteSCPGet(String host, String user,  String userpass, String file, String localdir,Integer port,String charSet) {
070                this.host = host;
071                this.user = user;
072                this.userpass=userpass;
073                this.file = file;
074                this.localdir = localdir;
075                this.port=port;
076                this.characterSet=charSet;
077        }
078
079        public String Result(SshConfig ac) throws IOException{
080                Object sshuser=ac.getConfig("USER");
081                Object sshpass=ac.getConfig("PASS");
082                Object sshkey=ac.getConfig("KEY");
083                Object sshkeypass=ac.getConfig("KEYPASS");
084                Object sshport=ac.getConfig("PORT");
085                String charSet=ac.getConfig("CHARACTERSET").toString();
086                
087                //System.out.println("----"+sshuser.toString());
088                if (user.equals("")) {
089                        user = sshuser.toString();
090                }
091                if (userpass.equals("")) {
092                        userpass = sshpass.toString();
093                }
094                if (port==0) {
095                        String sps=sshport.toString();
096                        if (sps.matches("[0-9]+")) {
097                                port=Integer.parseInt(sps);
098                        }
099                }
100                if (this.characterSet.length()==0) {
101                        if (charSet.length()>0) {
102                                this.characterSet=charSet;
103                        } else {
104                                this.characterSet="UTF-8";
105                        }
106                }
107
108                String hostname = host;
109                String username = user;
110                File keyfile = new File(sshkey.toString());
111                String keyfilePass = sshkeypass.toString();
112                        if (port==0){port=22; }
113                        Connection conn = new Connection(hostname,port);
114                        /* Now connect */
115                        conn.connect();
116                        /* Authenticate */
117                        boolean isAuthenticated=false;
118                        if (userpass.equals("")) {
119                                isAuthenticated = conn.authenticateWithPublicKey(username,
120                                                keyfile, keyfilePass);
121                        }else{
122                                isAuthenticated = conn.authenticateWithPassword(username,userpass);
123                        }
124
125                        if (isAuthenticated == false)
126                                throw new IOException("Authentication failed.");
127
128                        /* Create a session */
129                        SCPClient scp = conn.createSCPClient();
130                        if (characterSet!=null) {
131                                scp.setCharset(characterSet);
132                        }
133                        scp.get(file);
134                        conn.close();
135                        output = "File " + file + " should now be copied from " + host
136                                        + " to localdir: " + localdir + "<br>";
137
138                return output;
139        }
